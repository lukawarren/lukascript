fn factorial: int number

    // Check bounds first
    if number > 10
        return 0
    done

    if number is 1
        return 1
    done

    factorial(number - 1) -> ret
    return number * ret

done


for _ from 0 to 5

    input("Enter a number: ") -> number
    factorial(number) -> result
    print(number "! = " result)

done
